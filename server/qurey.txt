create table CUSTOMERS (
	ID int not null auto_increment primary key,
	NAME nvarchar(10),
	NOTE nvarchar(500),
    TEL nvarchar(15),
    DATE nvarchar(12),
    SCHOOL nvarchar(15),
    FEE int(100),
	REGISTER_ID nvarchar(20),
	REGISTER_DATE DATETIME DEFAULT now(),
	UPDATER_ID nvarchar(20),
	UPDATER_DATE DATETIME DEFAULT now()
);
